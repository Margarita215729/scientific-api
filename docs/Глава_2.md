import astropy.units as u

def load_vizier_data(ra_center: float, dec_center: float, 
                     radius_deg: float = 0.5, max_rows: int = 5000) -> pd.DataFrame:
    """
    Загружает данные из VizieR V/154 (SDSS DR16).
    
    Parameters:
        ra_center: Прямое восхождение центра поиска (градусы)
        dec_center: Склонение центра поиска (градусы)
        radius_deg: Радиус поиска (градусы)
        max_rows: Максимальное число объектов
    
    Returns:
        DataFrame с данными галактик
    """
    Vizier.TIMEOUT = 120
    v = Vizier(
        columns=["objID", "RA_ICRS", "DE_ICRS", "class", "zsp", 
                 "umag", "gmag", "rmag", "imag", "zmag"],
        row_limit=max_rows
    )
    coord = SkyCoord(ra=ra_center, dec=dec_center, unit="deg")
    result = v.query_region(coord, radius=radius_deg*u.deg, catalog="V/154")
    
    if result:
        return result[0].to_pandas()
    return pd.DataFrame()

## 2.4. Построение графов

### 2.4.1. Космологические графы: метод k-ближайших соседей

Для построения графов космической паутины используется метод k-ближайших соседей (k-NN) в комовинговых координатах. Преобразование из экваториальных координат и красного смещения выполняется по формулам:

$$D_c = \frac{c \cdot z}{H_0}$$

$$x = D_c \cdot \cos(\delta) \cdot \cos(\alpha)$$
$$y = D_c \cdot \cos(\delta) \cdot \sin(\alpha)$$
$$z_{coord} = D_c \cdot \sin(\delta)$$

где $\alpha$, $\delta$ — прямое восхождение и склонение в радианах, $z$ — красное смещение, $c = 299792.458$ км/с — скорость света, $H_0 = 70$ км/с/Мпк — постоянная Хаббла.

```python
def ra_dec_z_to_comoving(ra, dec, z, H0=70.0):
    """Преобразование в комовинговые координаты."""
    c = 299792.458  # km/s
    D_c = c * z / H0  # Mpc
    ra_rad, dec_rad = np.radians(ra), np.radians(dec)
    x = D_c * np.cos(dec_rad) * np.cos(ra_rad)
    y = D_c * np.cos(dec_rad) * np.sin(ra_rad)
    z_coord = D_c * np.sin(dec_rad)
    return np.column_stack([x, y, z_coord])

def build_knn_graph(df: pd.DataFrame, k: int = 15, 
                    max_nodes: int = 2000) -> nx.Graph:
    """Построение k-NN графа из космологических данных."""
    if len(df) > max_nodes:
        df = df.sample(n=max_nodes, random_state=42)
    
    coords = ra_dec_z_to_comoving(
        df["ra"].values, df["dec"].values, df["z"].values
    )
    
    nn = NearestNeighbors(n_neighbors=k+1, algorithm="ball_tree")
    nn.fit(coords)
    distances, indices = nn.kneighbors(coords)
    
    G = nx.Graph(system_type="cosmology", source="SDSS_REAL")
    for i in range(len(df)):
        G.add_node(i, ra=df.iloc[i]["ra"], 
                   dec=df.iloc[i]["dec"], z=df.iloc[i]["z"])
    
    for i in range(len(df)):
        for j_idx in range(1, k+1):
            j = indices[i, j_idx]
            if i < j:
                G.add_edge(i, j, weight=distances[i, j_idx])
    
    return G
```

### 2.4.2. Квантовые графы: решёточные модели

Для сравнительного анализа построены графы квантовых систем на основе моделей, описанных в главе 1:

#### Двумерный гармонический осциллятор

Потенциал определяется как:
$$V(x, y) = \frac{1}{2}(\omega_x^2 x^2 + \omega_y^2 y^2)$$

```python
def harmonic_oscillator_2d(grid_size: int = 40, 
                           omega_x: float = 1.0, 
                           omega_y: float = 1.0):
    """2D гармонический осциллятор."""
    x = np.linspace(-3, 3, grid_size)
    X, Y = np.meshgrid(x, x)
    V = 0.5 * (omega_x**2 * X**2 + omega_y**2 * Y**2)
    return X, Y, V
```

#### Потенциальная яма

$$V(x, y) = \begin{cases} -V_0, & |x| < w/2 \text{ и } |y| < w/2 \\ V_{barrier}, & \text{иначе} \end{cases}$$

```python
def potential_well_2d(grid_size: int = 40, 
                      depth: float = 10.0, 
                      width: float = 2.0):
    """2D потенциальная яма."""
    x = np.linspace(-3, 3, grid_size)
    X, Y = np.meshgrid(x, x)
    V = np.where(
        (np.abs(X) < width/2) & (np.abs(Y) < width/2), 
        -depth, 20.0
    )
    return X, Y, V
```

#### Построение решёточного графа

```python
def build_lattice_graph(X, Y, V) -> nx.Graph:
    """Построение решёточного графа квантовой системы."""
    gs = X.shape[0]
    G = nx.Graph(system_type="quantum")
    
    # Добавление узлов
    for i in range(gs):
        for j in range(gs):
            G.add_node(i*gs+j, x=X[i,j], y=Y[i,j], potential=V[i,j])
    
    # Добавление рёбер (соседние узлы решётки)
    for i in range(gs):
        for j in range(gs):
            for di, dj in [(-1,0), (1,0), (0,-1), (0,1)]:
                ni, nj = i+di, j+dj
                if 0 <= ni < gs and 0 <= nj < gs:
                    if i*gs+j < ni*gs+nj:
                        weight = 1 + 0.1*abs(V[i,j] - V[ni,nj])
                        G.add_edge(i*gs+j, ni*gs+nj, weight=weight)
    return G
```

### 2.5.2. Спектральные характеристики

На основе лапласиана графа $L = D - A$ вычисляются спектральные признаки:

```python
def extract_spectral_features(G: nx.Graph) -> Dict[str, float]:
    """Извлечение спектральных характеристик графа."""
    L = nx.laplacian_matrix(G).toarray()
    eigenvalues = np.sort(np.linalg.eigvalsh(L))
    
    features = {
        "spectral_gap": eigenvalues[1] if len(eigenvalues) > 1 else 0,
    }
    for i in range(min(5, len(eigenvalues))):
        features[f"eig_{i}"] = eigenvalues[i]
    
    return features
```

### 2.5.3. Итоговый набор признаков

```python
def extract_features(G: nx.Graph) -> Dict[str, Any]:
    """Полное извлечение признаков графа."""
    n, m = G.number_of_nodes(), G.number_of_edges()
    if n == 0:
        return {"n_nodes": 0}
    
    degrees = np.array([d for _, d in G.degree()])
    
    features = {
        "n_nodes": n,
        "n_edges": m,
        "density": nx.density(G),
        "avg_degree": degrees.mean(),
        "std_degree": degrees.std(),
        "max_degree": degrees.max(),
        "avg_clustering": nx.average_clustering(G),
        "transitivity": nx.transitivity(G),
        "n_components": nx.number_connected_components(G)
    }
    
    # Спектральные характеристики
    if n < 2000:
        L = nx.laplacian_matrix(G).toarray()
        eigs = np.sort(np.linalg.eigvalsh(L))
        features["spectral_gap"] = eigs[1] if len(eigs) > 1 else 0
        for i in range(min(5, len(eigs))):
            features[f"eig_{i}"] = eigs[i]
    
    return features
```

**Итого: 16 признаков на граф.**

## 2.6. Сравнительный анализ характеристик графов

### 2.6.1. Различия между космологическими и квантовыми графами

Анализ извлечённых признаков выявил существенные структурные различия между двумя типами систем:

| Характеристика | Космологические | Квантовые |
|----------------|-----------------|-----------|
| Плотность | 0.025–0.038 | 0.0024 |
| Средняя степень | 10–15 | 3.9 |
| Кластеризация | 0.64–0.67 | 0.0 |
| Транзитивность | 0.64–0.65 | 0.0 |
| Компонент связности | 1 | 1 |
| Спектральный зазор | 5.8–20.7 | 0.006 |

**Ключевые наблюдения:**

1. **Кластеризация**: Космологические графы демонстрируют высокий коэффициент кластеризации (~0.65), отражающий склонность галактик образовывать группы и скопления. Квантовые решётки имеют нулевую кластеризацию из-за отсутствия треугольников в регулярной 4-связной топологии.

2. **Степень вершин**: Космологические графы имеют переменную степень (10–19), соответствующую неоднородности распределения галактик. Квантовые графы имеют почти постоянную степень (~4) из-за регулярной структуры.

3. **Спектральный зазор**: Значительно больший спектральный зазор в космологических графах указывает на более «жёсткую» связность по сравнению с квантовыми решётками.
